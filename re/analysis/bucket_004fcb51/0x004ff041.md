### Flow (simplified)

```
loop:
  /* 1. Reduce-default-if-set */
  if yydefred[state] == 0:
     if yychar < 0: yychar = yylex()  /* FUN_004fd0d3 */
     /* 2. Shift attempt */
     if yysindex[state] != 0:
        n = yysindex[state] + yychar
        if 0 <= n < YYTABLESIZE and yycheck[n] == yychar:
            shift: push yytable[n], reset yychar
            goto loop
     /* 3. Reduce attempt */
     if yyrindex[state] != 0:
        n = yyrindex[state] + yychar
        if 0 <= n < YYTABLESIZE and yycheck[n] == yychar:
            reduce_rule = yytable[n]; goto action
     /* 4. Error */
     if yyerrflag == 0: yyerror("syntax error"); yynerrs++
     enter "error recovery" — pop states until "error" token can shift
  else:
     reduce_rule = yydefred[state]
  action:
     call FUN_004febf6(reduce_rule, yylen[reduce_rule])
     pop yylen[reduce_rule] states; goto yygindex/yydgoto for next state
     push new state; goto loop
```

### Error handling

- "syntax error" — first error per token; calls `FUN_004fd405(DAT_007da4e8, "syntax error")`.
