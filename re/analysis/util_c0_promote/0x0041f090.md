```c
word = DAT_0063dc74[param_1 * 0x2ac]   // 0x2ac = 684 bytes per player
if (param_2 != NULL)  *param_2 = word & 0x10   // bit 4
if (param_3 != NULL)  *param_3 = word & 0x20   // bit 5
```
